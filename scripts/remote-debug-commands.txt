# OpenUSP Linux Remote Debugging Commands
# Copy and run these commands on the Linux machine

# Method 1: SSH and check status
ssh -i ~/.ssh/openusp-demo.pem ubuntu@openusp.plumepoc.net "cd /home/ubuntu/projects/openusp && git pull && make status"

# Method 2: SSH and restart services
ssh -i ~/.ssh/openusp-demo.pem ubuntu@openusp.plumepoc.net "cd /home/ubuntu/projects/openusp && git pull && make dev-reset"

# Method 3: Interactive SSH session
ssh -i ~/.ssh/openusp-demo.pem ubuntu@openusp.plumepoc.net

# Then on the remote machine:
cd /home/ubuntu/projects/openusp
git pull
make status

# Manual restart if needed:
make force-stop
make infra-up
make run-services

# Check status:
make status-quick
